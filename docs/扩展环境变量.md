# 扩展环境变量

## 修改config-overrides.js
```js
const webpack = require('webpack');

module.exports = function override(config, env) {
  const version = require('./package.json').version;

  // 注意: version需要做一次JSON.stringify转换
  config.plugins.push(
    new webpack.DefinePlugin({
      'process.env': { VERSION: JSON.stringify(version) },
    })
  );

  return config;
};
```

## 示例：动态修改启动页面
* 有的内容直接在浏览器中调试更加方便
* 只用通过参数就能调试目标页面
### 步骤
1. 在`config-overrides.js`中检测参数
```js
// 注入process.env.START_PAGE
const { injectBabelPlugin } = require('react-app-rewired');
const webpack = require('webpack');
const program = require('commander');
const version = require('./package.json').version;

program
  .version(version)
  .option('-s, --start <name>', 'start page')
  .option('-w, --web', 'show at web')
  .parse(process.argv);

module.exports = function override(config, env) {
  if (!program.web) config.target = 'electron-renderer';

  config = injectBabelPlugin('transform-decorators-legacy', config);

  config.plugins.push(
    new webpack.DefinePlugin({
      'process.env': {
        VERSION: JSON.stringify(version),
        START_PAGE: JSON.stringify(program.start),
      },
    })
  );

  return config;
};
```
2. 修改入口文件`src\index.js`
```js
// 动态加载启动页面

import React from 'react';
import ReactDOM from 'react-dom';
import './index.css';
import registerServiceWorker from './registerServiceWorker';

async function start() {
  const startPage = process.env.START_PAGE || 'App';
  try {
    const StartComponent = (await import(`./${startPage}`)).default;
    ReactDOM.render(<StartComponent />, document.getElementById('root'));
  } catch (e) {
    const App = (await import('./App')).default;
    ReactDOM.render(<App />, document.getElementById('root'));
  }
}
start();

registerServiceWorker();
```
3. 使用
```bash
# 在浏览器中调试
yarn start -s WebTest -w

# 在Electron中调试
yarn start -s WebTest
```
